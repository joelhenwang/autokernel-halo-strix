"""Standalone Newton-Schulz throughput micro-benchmark.

Runs the 5-step Polar Express iteration on a fixed random 2D matrix,
reports time per call for fp32 and fp16 compute. Provides a hard
reference for NS performance on gfx1151 independent of PyTorch training.

Usage:
    python scripts/bench_newton_schulz.py
    python scripts/bench_newton_schulz.py --iters 2000 --shape 768 2816
"""
import argparse
import time
import torch
from halo_training.muon import zeropower_via_polar_express


def bench(shape, iters, dtype, device="cuda"):
    m = torch.randn(*shape, device=device, dtype=torch.float32)
    # warmup
    for _ in range(5):
        _ = zeropower_via_polar_express(m, steps=5, dtype=dtype)
    torch.cuda.synchronize()
    t0 = time.time()
    for _ in range(iters):
        _ = zeropower_via_polar_express(m, steps=5, dtype=dtype)
    torch.cuda.synchronize()
    return (time.time() - t0) / iters


def main():
    p = argparse.ArgumentParser()
    p.add_argument("--iters", type=int, default=500)
    p.add_argument("--shape", type=int, nargs=2, default=[768, 2816],
                   help="Matrix shape (rows cols) for Newton-Schulz")
    args = p.parse_args()

    print(f"NS shape: {args.shape}  iters: {args.iters}")
    results = {}
    for dtype in (torch.float32, torch.float16):
        t = bench(args.shape, args.iters, dtype)
        results[str(dtype).split(".")[-1]] = t
        print(f"  {dtype}: {t*1000:.3f} ms/call")

    print(f"fp16 speedup vs fp32: {results['float32']/results['float16']:.2f}x")

    import json
    out = {"shape": args.shape, "iters": args.iters, "results_ms_per_call": {k: v * 1000 for k, v in results.items()}}
    with open("docs/perf/normuon-ns-benchmark.json", "w") as f:
        json.dump(out, f, indent=2)
    print(f"\nWritten docs/perf/normuon-ns-benchmark.json")


if __name__ == "__main__":
    main()
```

from collections import defaultdict

def _normuon_step(self, group):
    lr = group["lr"]
    mu = group["momentum"]
    nesterov = group["nesterov"]
    ns_steps = group["ns_steps"]
    wd = group["weight_decay"]

    # Group 2D params by shape so same-shape groups can share a batched NS.
    shape_groups: dict = defaultdict(list)
    for p in group["params"]:
        if p.grad is None:
            continue
        if p.ndim != 2:
            # Non-2D safety fallback (unchanged from prior impl)
            p.data.add_(p.grad, alpha=-lr)
            continue
        shape_groups[tuple(p.shape)].append(p)

    for shape, plist in shape_groups.items():
        if len(plist) == 1:
            self._normuon_single(plist[0], lr, mu, nesterov, ns_steps, wd)
        else:
            self._normuon_batched(plist, lr, mu, nesterov, ns_steps, wd)

def _normuon_batched(self, params, lr, mu, nesterov, ns_steps, wd):
    """Stack same-shape params into [K, M, N] and run batched NS."""
    # Stack gradients and (per-param) momentum buffers
    grads = torch.stack([p.grad for p in params], dim=0)  # [K, M, N]
    bufs = []
    for p in params:
        state = self.state[p]
        if "momentum_buffer" not in state:
            state["momentum_buffer"] = torch.zeros_like(p.grad)
        bufs.append(state["momentum_buffer"])
    buf = torch.stack(bufs, dim=0)

    buf.mul_(mu).add_(grads)
    if nesterov:
        m = grads + buf * mu
    else:
        m = buf.clone()

    # Batched Newton-Schulz
    m_orth = _batched_newton_schulz_polar_express(
        m, steps=ns_steps, dtype=self.ns_dtype
    )  # [K, M, N] orthogonalized

    # Batched neuron-wise normalize (respecting min-dim gate)
    if self.neuron_wise_norm and (
        self.neuron_norm_min_dim == 0
        or min(m_orth.shape[-2:]) >= self.neuron_norm_min_dim
    ):
        # per-batch-item per-row L2 norm
        norms = m_orth.norm(dim=-1, keepdim=True).clamp_min(1e-6)
        m_orth = m_orth / norms

    # muP-style scaling (unchanged)
    scale = max(m_orth.shape[-2], m_orth.shape[-1]) ** 0.5 * 0.2
    m_orth.mul_(scale)

    # Unstack and apply per-param cautious WD + update
    for i, p in enumerate(params):
        m_i = m_orth[i]
        grad_i = grads[i]
        if wd > 0:
            if self.cautious_wd:
                mask = _cautious_wd_mask(grad_i, p.data)
                p.data.sub_(p.data * (lr * wd) * mask.to(p.dtype))
            else:
                p.data.mul_(1 - lr * wd)
        p.data.add_(m_i, alpha=-lr)

        # Propagate the updated momentum buffer back (in-place mul_+add_ on
        # state["momentum_buffer"] via buf view doesn't actually persist —
        # we must copy_ the results back)
        self.state[p]["momentum_buffer"].copy_(buf[i])
```

Also add `_batched_newton_schulz_polar_express` helper:

```python
def _batched_newton_schulz_polar_express(G, steps=5, dtype=None):
    """Batched NS over leading dim K: input [K, M, N], output [K, M, N]."""
    assert G.ndim == 3, f"Batched NS requires 3D input; got {tuple(G.shape)}"
    X = G.to(dtype or torch.float32)
    # Handle orientation per batch item consistently: pick the one where M>=N
    # across the whole batch (assumes homogeneous shape, which batching requires).
    if X.shape[1] > X.shape[2]:
        X = X.transpose(-2, -1)
        transposed = True
    else:
        transposed = False
    # Per-batch-item normalization
    norm_per_item = X.flatten(start_dim=-2).norm(dim=-1, keepdim=True).unsqueeze(-1) * 1.02 + 1e-7
    X = X / norm_per_item

    # Reuse _PE_COEFFS from halo_training.muon
    from halo_training.muon import _PE_COEFFS
    for i in range(min(steps, len(_PE_COEFFS))):
        a, b, c = _PE_COEFFS[i]
        A = X @ X.transpose(-2, -1)         # [K, M, M]
        B = b * A + c * (A @ A)              # [K, M, M]
        X = a * X + B @ X                    # [K, M, N]

    if transposed:
        X = X.transpose(-2, -1)
    return X.to(G.dtype)
```

### Task D.1 — Kernel skeleton

**File:** `kernels/hip/fused_newton_schulz.py` (NEW)

Follow the existing `kernels/hip/fused_rope_gate_mul.py` template:

```python
"""Fused HIP Newton-Schulz iteration for NorMuon optimizer.

Runs all 5 Polar Express iterations in a single HIP kernel per call,
using half2 loads + wave32 exec + fp32 accumulators. Designed for
batched input of shape [K, M, N] where K is the number of same-shape
2D parameters being optimized together.
"""

from kernels.hip._compile import compile_hip

HIP_SRC = r"""
// ... HIP kernel ...
"""

_module = None

def _get_module():
    global _module
    if _module is None:
        _module = compile_hip(HIP_SRC, "fused_newton_schulz")
    return _module


def fused_newton_schulz(G, steps=5, dtype=torch.float16):
    """Batched NS via fused HIP kernel."""
    mod = _get_module()
    return mod.fused_newton_schulz(G, steps)
```

**Kernel design points:**
- Input: `[K, M, N]` fp16 tensor (stacked gradients/momentum)
- Output: `[K, M, N]` fp16 orthogonalized tensor
- Compile-time or runtime steps parameter (start with 5 hardcoded;
  extend to 3/4 variants only if the ablation shows benefit)
- One wave (32 threads) per matrix row for the matmul inner loops
- Shared memory for X @ X.T intermediates
- Use `_PE_COEFFS` as constant memory

**Exit criterion:** kernel compiles, runs on test input shape
`[8, 768, 2816]`, produces finite output.

**Test:** `scripts/test_sprint1_1_normuon.py::test_hip_ns_matches_pytorch`
— same random input shape, compare HIP vs PyTorch batched NS, assert
max abs diff < 1e-3 (looser tolerance than C1's 1e-5 because fp16).

**Rollback:** delete HIP files; `normuon.py` stays on the PyTorch batched
path. Add a `--ns-backend {pytorch, hip}` CLI flag to opt in; default pytorch.
